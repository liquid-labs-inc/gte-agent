// ---
// name: deep-research
// description: Multi-angle market/asset research with claim cross-checking and a cited synthesis
// ---
// Bundled deep-research workflow for the read-only GTE trading-data agent:
// 1. plan: derive distinct research angles for the question
// 2. research: fan out one agent per angle over the gte_* data tools
// 3. cross-check: every substantive claim is independently re-verified; claims that fail are dropped
// 4. synthesis: one agent writes the cited answer from the surviving claims
const question = typeof args === "string" ? args : (args && args.question) || ""
if (!question.trim()) return "deep-research needs a question. Usage: /deep-research <market or asset question>"

const ANGLES = 5
const VERIFIERS = 2

const planned = await phase("plan", () =>
  agent({
    prompt: [
      `Research question about GTE markets/assets: ${question}`,
      ``,
      `Propose exactly ${ANGLES} distinct research angles that together give broad coverage of this question`,
      `using only read-only GTE exchange data (e.g. order book and depth, recent trades, candles and price`,
      `action, open interest and funding, positions and account activity, liquidations, market metadata).`,
      `Respond with ONLY a JSON array of ${ANGLES} short angle descriptions, no other text.`,
    ].join("\n"),
    type: "general",
  }),
)

let angles = [question]
const plannedMatch = planned.text.match(/\[[\s\S]*\]/)
if (plannedMatch) {
  const parsed = JSON.parse(plannedMatch[0])
  if (Array.isArray(parsed) && parsed.length > 0) angles = parsed
}
log(`researching ${angles.length} angles`)

const findings = await phase("research", () =>
  map(
    angles,
    (angle) =>
      agent({
        prompt: [
          `You are a read-only GTE trading-data researcher.`,
          `Question: ${question}`,
          `Your angle: ${angle}`,
          ``,
          `Use the gte_* data tools to gather the snapshots relevant to your angle. Each gte_* result is`,
          `{ provenance, data }; provenance records the resolved environment, an ISO timestamp, the source,`,
          `and the query parameters. Snapshots are not streams — call a tool again when you need fresher data.`,
          `Extract the key factual, data-backed claims relevant to the question.`,
          `For every claim include the gte_* tool and the provenance timestamp it came from.`,
          `Respond with ONLY a JSON array of objects: [{ "claim": "...", "source": "gte_<tool> @ <iso-timestamp>" }].`,
          `Only include claims a snapshot you actually fetched supports.`,
        ].join("\n"),
        type: "general",
      }),
    { concurrency: 4 },
  ),
)

const claims = []
for (const finding of findings) {
  const findingMatch = finding.text.match(/\[[\s\S]*\]/)
  if (!findingMatch) continue
  const parsed = JSON.parse(findingMatch[0])
  if (!Array.isArray(parsed)) continue
  for (const item of parsed) {
    if (item && typeof item.claim === "string") claims.push({ claim: item.claim, source: item.source || "" })
  }
}
log(`cross-checking ${claims.length} claims`)
if (claims.length === 0) return "deep-research found no verifiable claims for: " + question

const votes = await phase("cross-check", () =>
  map(
    claims,
    (item) =>
      map(
        Array.from({ length: VERIFIERS }, (_, i) => i),
        (i) =>
          agent({
            prompt: [
              `Independently verify this data claim about GTE markets (verifier #${i + 1}):`,
              `CLAIM: ${item.claim}`,
              `STATED SOURCE: ${item.source}`,
              ``,
              `Re-fetch the relevant gte_* snapshot(s) yourself and check whether the claim is accurate and`,
              `currently supported by the data. Treat stale or contradicted figures as not supported.`,
              `Respond with ONLY one word on the first line: SUPPORTED, REFUTED, or UNCLEAR,`,
              `then a one-sentence justification on the second line.`,
            ].join("\n"),
            type: "general",
          }),
        { concurrency: VERIFIERS },
      ),
    { concurrency: 3 },
  ),
)

const surviving = []
for (let i = 0; i < claims.length; i++) {
  const verdicts = votes[i].map((vote) => (vote.text.trim().split(/\s/)[0] || "").toUpperCase())
  const supported = verdicts.filter((verdict) => verdict.startsWith("SUPPORTED")).length
  const refuted = verdicts.filter((verdict) => verdict.startsWith("REFUTED")).length
  if (supported > refuted) surviving.push(claims[i])
  else log(`dropped claim (votes ${supported}-${refuted}): ${claims[i].claim.slice(0, 80)}`)
}
log(`${surviving.length}/${claims.length} claims survived cross-checking`)
if (surviving.length === 0) return "deep-research could not cross-check any claims for: " + question

const report = await phase("synthesis", () =>
  agent({
    prompt: [
      `Write a research answer for this GTE markets/assets question: ${question}`,
      ``,
      `Use ONLY the following cross-checked claims. Cite the gte_* source for every figure you use.`,
      `Report exchange data, not trading advice — never recommend or preview a trade.`,
      ``,
      JSON.stringify(surviving, null, 2),
      ``,
      `Structure: a direct answer first, then supporting sections, then a "Sources" list of the gte_* snapshots.`,
      `Note explicitly where the data was thin, stale, or contested.`,
    ].join("\n"),
    type: "general",
    variant: "high",
  }),
)

return report.text
